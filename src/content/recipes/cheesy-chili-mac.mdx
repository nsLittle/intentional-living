ingredients:
  - "2 cups elbow macaroni"
  - "2/3 cup cheese powder"
  - "3 Tablespoons dry milk powder"
  - "1/4 cup taco seasoning mix"
  - "2 Tablespoons dry butter powder"
  - "1 teaspoon salt"
  - "1 teaspoon black pepper"
  - "1/4 cup dried corn"
  - "1/8 cup dried carrots"
  - "1/8 cup dried onions"

steps:

  - "Store or use as desired."

cookingPrepTime: "15 minutes"
cookingTime: "25 minutes"
cookingIngredients:
  - "1 Jar Cheesy Chili Mac Jar"
  - "1 Tablespoon vegetable oil"
  - "1 lb of ground protein, such as beef, chicken, or turkey"
  - "5 cups water"

cookingSteps:
  - "Heat oil in large stock pot over high heat."
  - "Saute protein until browned."
  - "Mix all remaining ingredients into the stock pot."
  - "Bring to a boil over high heat."
  - "Reduce heat to low; let simmer for 10 minutes or until noodles are tender."
  - "Season to taste."
  - "Serve with tortilla chips. Enjoy!"
